CREATE DATABASE PRD;
USE PRD;

CREATE TABLE USUARIOS(
ID MEDIUMINT AUTO_INCREMENT,
NOMBRE CHAR(255),
CONTRASENA CHAR(255),
SECTOR CHAR(255),
PRIMARY KEY(ID)
);
CREATE TABLE REGISTRO(
id MEDIUMINT AUTO_INCREMENT,
Nombre CHAR(255),
Telefono char(255),
Correo char(255),
CLAVE_ELECTOR CHAR(255),
CURP CHAR(255),
SECCION CHAR(255),
PRIMARY KEY(id)
)
;
